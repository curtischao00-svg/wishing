#!/usr/bin/env python3 """ 许愿池 — 本地局域网服务器 运行后，在同一 WiFi 下的手机和电脑都可以访问。 """ import http.server import socket import os import sys PORT = 8888 DIRECTORY = os.path.dirname(os.path.abspath(__file__)) def get_local_ip(): try: s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM) s.connect(("8.8.8.8", 80)) ip = s.getsockname()[0] s.close() return ip except Exception: return "127.0.0.1" class Handler(http.server.SimpleHTTPRequestHandler): def __init__(self, *args, **kwargs): super().__init__(*args, directory=DIRECTORY, **kwargs) def log_message(self, format, *args): # 简洁日志 print(f" {self.address_string()} → {args[0]}") if __name__ == "__main__": local_ip = get_local_ip() print() print(" 🌊 许愿池已启动！") print() print(f" 本机访问： http://localhost:{PORT}") print(f" 手机访问： http://{local_ip}:{PORT}") print() print(" 按 Ctrl+C 停止服务器") print() os.chdir(DIRECTORY) with http.server.HTTPServer(("0.0.0.0", PORT), Handler) as httpd: try: httpd.serve_forever() except KeyboardInterrupt: print("\n 服务器已停止。") sys.exit(0)
